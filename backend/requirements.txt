# Python dependencies for the Digital CookBook backend.
# We add libraries here one at a time as we need them.

# Calls the Apify Instagram scraper API over HTTP.
requests

# Loads secrets (API tokens) from the .env file into the environment.
python-dotenv

# Google Gemini SDK — parses captions into structured recipe data.
google-genai

# Web framework + ASGI server that expose the backend as an HTTP API.
fastapi
uvicorn[standard]

# Supabase client — reads/writes our hosted Postgres database.
supabase
